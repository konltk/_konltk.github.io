[docs]def test_asp():
    k = KltAsp()
    assert k.asp(text="국민대학교자연어처리연구실") == ['국민대학교', '자연어처리', '연구실']
    assert k.asp(text="국민대학교자연어처리연구실", split=False) == '국민대학교 자연어처리 연구실'
    assert k.asp(text="국민대학교자연어처리연구실", split=True) == ['국민대학교', '자연어처리', '연구실']
